-------for back---------

server {
    server_name staging-app.wellnessmahotsav.com;

    location / {
        proxy_pass http://127.0.0.1:3009;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

}

sudo ln -s /etc/nginx/sites-available/staging-app.wellnessmahotsav.com /etc/nginx/sites-enabled/staging-app.wellnessmahotsav.com

sudo certbot --nginx -d staging-app.wellnessmahotsav.com

sudo systemctl restart nginx

sudo systemctl status nginx

---------------------server config apache2--------------------------------

-------for back---------
nano /etc/apache2/sites-available/api.erosy.com.conf

<VirtualHost *:80>
    ServerName api.erosy.com

    # Proxy to Node.js app on port 3001
    ProxyPreserveHost On
    ProxyPass / http://127.0.0.1:3001/
    ProxyPassReverse / http://127.0.0.1:3001/

    # Optional security headers
    Header always set X-Content-Type-Options "nosniff"
    Header always set X-Frame-Options "SAMEORIGIN"
    Header always set X-XSS-Protection "1; mode=block"

    # Logs
    ErrorLog ${APACHE_LOG_DIR}/api.erosy.error.log
    CustomLog ${APACHE_LOG_DIR}/api.erosy.access.log combined
</VirtualHost>

sudo a2enmod proxy proxy_http headers
sudo a2ensite api.erosy.com.conf
sudo systemctl reload apache2

-------for front---------
sudo nano /etc/apache2/sites-available/admin.erosy.com.conf

<VirtualHost *:80>
    ServerAdmin info@erosy.com
    ServerName admin.erosy.com
    DocumentRoot /var/www/html/admin.erosy.com

    ErrorLog ${APACHE_LOG_DIR}/admin.erosy.error.log
    CustomLog ${APACHE_LOG_DIR}/admin.erosy.access.log combined

    <Directory /var/www/html/admin.erosy.com>
        AllowOverride All
        Require all granted
    </Directory>
</VirtualHost>

sudo a2ensite admin.erosy.com.conf
sudo systemctl reload apache2
